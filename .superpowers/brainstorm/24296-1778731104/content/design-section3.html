Database: New table store_theme_settings

create table store_theme_settings (
  id uuid primary key default gen_random_uuid(),
  store_id uuid not null references stores(id) on delete cascade,
  theme_name text not null default 'classic', -- 'classic' | 'brio'

  -- Hero
  hero_headline text,
  hero_subtitle text,
  hero_button_text text,
  hero_image_url text,

  -- Story
  story_text text,

  -- Featured items (array of product ids)
  featured_product_ids uuid[],

  -- Contact
  contact_phone text,
  contact_address text,
  contact_hours text,
  google_maps_url text,

  created_at timestamp default now(),
  updated_at timestamp default now(),

  unique(store_id)
);

API Endpoints

  • GET /api/v1/public/stores/:slug/theme — public read of theme settings
  • PUT /api/v1/merchant/theme — merchant update theme settings
